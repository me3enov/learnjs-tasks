4.4.1 Using "this" in object literal

importance: 5

Tasks:

Here the function makeUser returns an object.

What is the result of accessing its ref? Why? 

function makeUser() {
  return {
    name: "John",
    ref: this
  };
}

let user = makeUser();

alert( user.ref.name ); // What's the result?

Answer:

Inside scripts.js : 

'use strict';

function makeUser() {
  return {
    name: "John",
    ref: this
  };
}

let user = makeUser();

// alert( user.ref.name ); // => Error: Cannot read property 'name' of undefined

// => Better

function makeUser2() {
  return {
    name: "John",
    ref() {
      return this;
    }
  };
}

let user2 = makeUser2();

alert( user2.ref().name );
