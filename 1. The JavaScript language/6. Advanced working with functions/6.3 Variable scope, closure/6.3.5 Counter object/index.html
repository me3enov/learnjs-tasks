6.3.5 Counter object

importance: 5

Tasks:

Here a counter object is made with the help of the constructor function.

Will it work? What will it show? 

function Counter() {
  let count = 0;

  this.up = function() {
    return ++count;
  };
  this.down = function() {
    return --count;
  };
}

let counter = new Counter();

alert( counter.up() ); // ?
alert( counter.up() ); // ?
alert( counter.down() ); // ?

Answer:

Inside scripts.js : 

'use strict';

function Counter() {
  let count = 0;

  this.up = function() {
    return ++count;
  };
  this.down = function() {
    return --count;
  };
}

let counter = new Counter();

alert( counter.up() ); // => 1
alert( counter.up() ); // => 2
alert( counter.down() ); // => 1
